🛡️ قواعد الأمان المطلوبة (Firestore Rules)

تأكد أن قواعد Firestore الخاصة بمجموعة config تسمح بالقراءة فقط للمستخدمين المسجلين: 

match /config/{docId} {
  // القراءة: مسموح لأي مستخدم مسجّل
  allow read: if request.auth != null;
  // الكتابة: مقيّدة بالمدير فقط عبر صفحة الإعداد هذه
  allow write: if request.auth.token.email == "ADMIN_EMAIL_HERE";
}
